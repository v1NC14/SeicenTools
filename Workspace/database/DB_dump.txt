TABLE Utente(
	id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
	nome VARCHAR(100) NOT NULL,
	hashpwd VARCHAR(255) NOT NULL,
	email VARCHAR(50) NOT NULL UNIQUE,
	ruolo VARCHAR(20) NOT NULL,
);


TABLE Prodotto(
	id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
	nome VARCHAR(100) NOT NULL,
	categoria VARCHAR(50) NOT NULL,
	descrizione VARCHAR(255),
	prezzo DECIMAL(10, 2) NOT BULL,
	imgPath VARCHAR(255),
	disponibilita INT NOT NULL
);

TABLE Pagamento(
	id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
	id_ordine INT NOT NULL,
	numCarta VARCHAR(12) NOT NULL,
	dataPagamento TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	
	FOREIGN KEY (id_ordine) REFERENCES Ordine(id) ON UPDATE CASCADE ON UPDATE DELETE CASCADE	
);

TABLE Ordine(
	id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
	id_utente INT NOT NULL,
	tot DECIMAL (10, 2),
	qta IN NOT NULL,
	dataCreazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP
	indirizzoConsegna VARCHAR(255) NOT NULL,
	FOREIGN KEY (id_utente) REFERENCES Utente(id) ON UPDATE CASCADE ON DELETE CASCADE
);

TABLE ProdottiOrdinati(
	id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
	qta INT NOT NULL,
	id_ordine INT NOT NULL,
	id_utente INT NOT NULL,

	FOREIGN KEY (id_utente) REFERENCES Utente(id) ON UPDATE CASCADE ON DELETE CASCADE,

	FOREIGN KEY (id_ordine REFERENCES Ordine(id) ON UPDATE CASCADE ON DELETE CASCADE
);
